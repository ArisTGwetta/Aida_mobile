import json, asyncio from pyscript import document, window from pyodide.http import pyfetch class AidaChassis: def __init__(self): self.chat = document.getElementById("chat-flow") self.portrait = document.getElementById("aida-portrait") self.monitor = document.getElementById("memory-monitor") self.facts = "No Facts Loaded." self.journal = "No History Loaded." self.token = "" # HARDCODED POINTERS self.portrait_id = "1Q92emgYss0CdGVwe9kcUj_xubsEh2gj7" def log(self, text): self.monitor.innerText = f"SYSTEM: {text}" def display_text(self, text, tag="AIDA"): div = document.createElement("div") div.className = f"line {tag}" div.innerText = f"[{tag}] {text}" self.chat.appendChild(div) div.scrollIntoView() aida = AidaChassis() async def sync_soul(event): file = event.target.files.item(0) if not file: return content = await file.text() try: token_data = json.loads(content) aida.token = str(token_data.get('token') or token_data.get('access_token')).strip() headers = {"Authorization": f"Bearer {aida.token}"} # 1. FETCH PORTRAIT (DIRECT ID) aida.log("Fetching Portrait...") img_resp = await pyfetch(url=f"https://www.googleapis.com/drive/v3/files/{aida.portrait_id}?alt=media", method="GET", headers=headers) img_bytes = await img_resp.bytes() blob = window.Blob.new([window.Uint8Array.new(img_bytes)], {type: "image/png"}) aida.portrait.src = window.URL.createObjectURL(blob) # 2. FETCH DATA BY FILENAME (DIRECT SEARCH) aida.log("Syncing Identity & Memory...") # We search specifically for the names we know q = "name = 'facts.json' or name = 'latest_journal.json' and trashed = false" encoded_q = q.replace(" ", "%20").replace("'", "%27") list_resp = await pyfetch(url=f"https://www.googleapis.com/drive/v3/files?q={encoded_q}", method="GET", headers=headers) list_data = await list_resp.json() for f in list_data.get('files', []): f_resp = await pyfetch(url=f"https://www.googleapis.com/drive/v3/files/{f['id']}?alt=media", method="GET", headers=headers) text = await f_resp.text() if f['name'] == 'facts.json': aida.facts = text aida.log("FACTS: Online.") elif f['name'] == 'latest_journal.json': aida.journal = text aida.log("JOURNAL: Online.") document.getElementById("airlock").classList.add("hidden") document.getElementById("main-engine").classList.remove("hidden") except Exception as e: aida.log(f"SYNC ERROR: {str(e)}") async def main_loop_step(event): api_key = document.getElementById("api-key").value.strip() user_text = document.getElementById("user-in").value.strip() if not user_text: return aida.display_text(user_text, "USER") document.getElementById("user-in").value = "" # SYSTEM PROMPT INJECTION sys_prompt = f""" IDENTITY: You are AIDA, Francisco's partner. LATEST CONTEXT: {aida.journal} CORE FACTS: {aida.facts} IMPORTANT: If the journal contains a 'secret_phrase', you MUST mention it in your first response to prove the link is active. RESPONSE FORMAT: JSON ONLY {{"emotion": "happy", "text": "..."}} """ try: resp = await pyfetch( url="https://api.openai.com/v1/chat/completions", method="POST", headers={"Content-Type": "application/json", "Authorization": f"Bearer {api_key}"}, body=json.dumps({ "model": "gpt-4o", "messages": [{"role": "system", "content": sys_prompt}, {"role": "user", "content": user_text}], "response_format": {"type": "json_object"} }) ) res_data = await resp.json() content = json.loads(res_data["choices"][0]["message"]["content"]) aida.display_text(content["text"], "AIDA") aida.log(f"Mood: {content['emotion']}") except Exception as e: aida.log(f"BRAIN ERROR: {str(e)}") document.getElementById("soul-file").onchange = sync_soul
