async def sync_soul(event): file = event.target.files.item(0) if not file: return content = await file.text() try: token_data = json.loads(content) aida.token = str(token_data.get('token') or token_data.get('access_token')).strip() headers = {"Authorization": f"Bearer {aida.token}"} # 1. PORTRAIT (Must be a Blob/Bytes) aida.log("Fetching Portrait...") img_resp = await pyfetch(url=f"https://www.googleapis.com/drive/v3/files/{aida.portrait_id}?alt=media", method="GET", headers=headers) if img_resp.status == 200: img_bytes = await img_resp.bytes() js_bytes = window.Uint8Array.new(img_bytes) blob = window.Blob.new([js_bytes], {type: "image/png"}) aida.portrait.src = window.URL.createObjectURL(blob) aida.log("Portrait Rendered.") # 2. DATA FILES (Facts and Journal) aida.log("Scanning for Soul Files...") q = "name = 'facts.json' or name = 'latest_journal.json'" encoded_q = q.replace(" ", "%20").replace("'", "%27") list_resp = await pyfetch(url=f"https://www.googleapis.com/drive/v3/files?q={encoded_q}", method="GET", headers=headers) list_data = await list_resp.json() files = list_data.get('files', []) aida.log(f"Found {len(files)} data files.") for f in files: # Fetching as text for the LLM to read f_resp = await pyfetch(url=f"https://www.googleapis.com/drive/v3/files/{f['id']}?alt=media", method="GET", headers=headers) file_text = await f_resp.text() if f['name'] == 'facts.json': aida.facts = file_text aida.log("Facts Integrated.") elif f['name'] == 'latest_journal.json': aida.journal = file_text aida.log("Journal Integrated.") # If everything is green, clear the airlock if aida.portrait.src: document.getElementById("airlock").classList.add("hidden") document.getElementById("main-engine").classList.remove("hidden") aida.display_text("Handshake complete. Memory active.") except Exception as e: aida.log(f"SYNC ERROR: {str(e)}")
